	C <- 100000
	# Truco de los ceros
	for (i in 1:N){
		zY[i] <- 0
		zY[i] ~ dpois(phi[i])
		fY[i] <- (alphaY / sY) * pow(y[i]/sY , -alphaY - 1) * exp(-pow(y[i]/sY , -alphaY))
		phi[i] <- -log(fY[i]) + C
	}
	# Aprioris PM25
	alphaY ~  dunif(0,10)
	sY ~ dunif(0,150)
}
